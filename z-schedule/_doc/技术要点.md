## 1. 调度算法

### 1.1 Cron表达式解析

**Quartz Cron 表达式格式：**
```
秒(0-59) 分(0-59) 时(0-23) 日(1-31) 月(1-12) 周(1-7) 年(可选)
```

**核心实现类：**
```java
// Cron表达式解析器
public class CronExpression {
    private final String cronExpression;
    private final TreeSet<Integer> seconds;
    private final TreeSet<Integer> minutes;
    private final TreeSet<Integer> hours;
    private final TreeSet<Integer> daysOfMonth;
    private final TreeSet<Integer> months;
    private final TreeSet<Integer> daysOfWeek;

    /**
     * 获取下一个有效时间
     */
    public Date getNextValidTimeAfter(Date afterTime) {
        // 从afterTime开始递增，找到匹配Cron的下一个时间点
        Calendar calendar = Calendar.getInstance();
        calendar.setTime(afterTime);
        calendar.add(Calendar.SECOND, 1); // 从下一秒开始

        boolean found = false;
        while (!found && calendar.get(Calendar.YEAR) < 2099) {
            found = isMatch(calendar);
            if (!found) {
                calendar.add(Calendar.SECOND, 1);
            }
        }
        return found ? calendar.getTime() : null;
    }
}
```

**常见Cron表达式：**
| 描述 | 表达式 |
|------|--------|
| 每秒执行 | `* * * * * ?` |
| 每分钟执行 | `0 * * * * ?` |
| 每小时执行 | `0 0 * * * ?` |
| 每天凌晨2点 | `0 0 2 * * ?` |
| 每周一上午9点 | `0 0 9 ? * MON` |
| 每月1号上午10点 | `0 0 10 1 * ?` |

### 1.2 时间轮算法 (Hashed Wheel Timer)

**适用场景：** 精确到毫秒级的延迟任务调度

**实现原理：**
```
┌─────────────────────────────────────────────────────────────┐
│                     时间轮结构 (Wheel)                        │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  tickDuration = 100ms  (每tick代表100毫秒)                  │
│  ticksPerWheel = 512     (一轮512个槽位)                     │
│  maxDelay = 100ms * 512 = 51.2秒 (单层最大延迟)              │
│                                                             │
│  ┌────┬────┬────┬────┬────┬────┬────┬────┬────┬────┐    │
│  │ 0  │ 1  │ 2  │ 3  │ 4  │ 5  │ ...│507 │508 │509 │... │    │
│  └────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┘    │
│    │                                                       │    │
│    ▼                                                       ▼    │
│  ┌─────────┐                                             ┌─────────┐│
│  │Task-1   │                                             │Task-N   ││
│  │delay=200ms│                                           │delay=51s││
│  │(round=0) │                                           │(round=1)││
│  └─────────┘                                             └─────────┘│
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

**Java实现：**
```java
public class HashedWheelTimer {
    private final long tickDuration;  // 每个tick的时间跨度
    private final int ticksPerWheel;  // 轮盘大小
    private final AtomicInteger wheelCursor = new AtomicInteger();
    private final Set<Timeout>[] wheel;
    private final Queue<Timeout> waitingForRounds = new ConcurrentLinkedQueue<>();

    public HashedWheelTimer(long tickDuration, TimeUnit unit, int ticksPerWheel) {
        this.tickDuration = unit.toNanos(tickDuration);
        this.ticksPerWheel = ticksPerWheel;
        this.wheel = new Set[ticksPerWheel];
        for (int i = 0; i < ticksPerWheel; i++) {
            wheel[i] = Collections.newSetFromMap(new ConcurrentHashMap<>());
        }
        start();
    }

    public Timeout newTimeout(TimerTask task, long delay, TimeUnit unit) {
        long deadline = System.nanoTime() + unit.toNanos(delay);
        long ticks = Math.max(unit.toNanos(delay) / tickDuration, 1);

        HashedWheelTimeout timeout = new HashedWheelTimeout(task, deadline, ticks);

        if (ticks == 0) {
            // 立即执行
            timeout.expire();
        } else if (ticks < ticksPerWheel) {
            // 本轮执行
            int stopIndex = (int) (wheelCursor.get() + ticks) % ticksPerWheel;
            wheel[stopIndex].add(timeout);
        } else {
            // 多轮后执行
            timeout.setRemainingRounds(ticks / ticksPerWheel);
            waitingForRounds.offer(timeout);
        }

        return timeout;
    }

    private void start() {
        Thread workerThread = new Thread(() -> {
            long startTime = System.nanoTime();
            while (!shutdown) {
                long deadline = waitForNextTick();
                if (deadline > 0) {
                    int idx = (int) (wheelCursor.getAndIncrement() % ticksPerWheel);
                    processExpiredTimeouts(idx);
                    transferTimeoutsToWheel();
                }
            }
        }, "HashedWheelTimerWorker");
        workerThread.start();
    }
}
```

```java
            TimeUnit.SECONDS,

                .build(),
            new ThreadPoolExecutor.DiscardPolicy()  // 拒绝策略：静默丢弃

     * 处理执行器上报的执行结果
     */
    @Bean("callbackThreadPool")
    public ThreadPoolExecutor callbackThreadPool() {
        return new ThreadPoolExecutor(
            5,
            50,
            60L,
            TimeUnit.SECONDS,
            new SynchronousQueue<>(),  // 直接提交队列
            new ThreadFactoryBuilder()
                .setNameFormat("zjob-callback-%d")
                .build(),
```

```java
@Component
public class ThreadPoolMonitor {

    @Autowired
    @Qualifier("fastTriggerPool")
    private ThreadPoolExecutor fastTriggerPool;

    @Autowired
    @Qualifier("slowTriggerPool")
    private ThreadPoolExecutor slowTriggerPool;

    /**
     * 定时上报线程池指标
     */
    @Scheduled(fixedRate = 60000) // 每分钟上报一次
    public void reportMetrics() {
        reportPoolMetrics("fast", fastTriggerPool);
        reportPoolMetrics("slow", slowTriggerPool);
    }

    private void reportPoolMetrics(String poolName, ThreadPoolExecutor pool) {
        // 核心指标
        int corePoolSize = pool.getCorePoolSize();
        int maximumPoolSize = pool.getMaximumPoolSize();
        int activeCount = pool.getActiveCount();
        int poolSize = pool.getPoolSize();
        long completedTaskCount = pool.getCompletedTaskCount();
        long taskCount = pool.getTaskCount();
        int queuedTasks = pool.getQueue().size();
        int remainingCapacity = pool.getQueue().remainingCapacity();

        // 计算关键指标
        double activeRatio = (double) activeCount / corePoolSize * 100;
        double queueUsage = (double) queuedTasks / (queuedTasks + remainingCapacity) * 100;
        long pendingTasks = taskCount - completedTaskCount;

        // 上报到监控系统 (Prometheus/Micrometer)
        Metrics.gauge("zjob.threadpool.active.ratio", poolName, activeRatio);
        Metrics.gauge("zjob.threadpool.queue.usage", poolName, queueUsage);
        Metrics.gauge("zjob.threadpool.pending.tasks", poolName, pendingTasks);

        // 告警判断
        if (activeRatio > 90) {
            alarmService.sendAlert(poolName + "线程池活跃线程接近上限: " + activeRatio + "%");
        }
        if (queueUsage > 80) {
            alarmService.sendAlert(poolName + "线程池队列使用率过高: " + queueUsage + "%");
        }
    }
}
```

```java
/**
 * HTTP客户端封装 (基于OkHttp)

@Component
public class HttpClient {

    private final OkHttpClient client;

    public HttpClient() {
        this.client = new OkHttpClient.Builder()
            .connectTimeout(3, TimeUnit.SECONDS)      // 连接超时
            .readTimeout(10, TimeUnit.SECONDS)        // 读取超时
            .writeTimeout(10, TimeUnit.SECONDS)       // 写入超时
            .connectionPool(new ConnectionPool(      // 连接池
                50,     // 最大空闲连接数
                5,      // 存活时间(分钟)
                TimeUnit.MINUTES
            ))
            .protocols(Arrays.asList(Protocol.HTTP_2, Protocol.HTTP_1_1))
            .build();
    }

    /**
     * 发送任务触发请求到执行器
     */
    public ReturnT<String> sendTriggerRequest(String address, TriggerParam triggerParam) {
        String url = address + "/run";

        try {
            // 序列化参数
            String json = JSON.toJSONString(triggerParam);
            RequestBody body = RequestBody.create(
                MediaType.parse("application/json; charset=utf-8"),
                json
            );

            Request request = new Request.Builder()
                .url(url)
                .post(body)
                .header("X-Job-Access-Token", generateToken(triggerParam))
                .build();

            // 发送请求
            try (Response response = client.newCall(request).execute()) {
                if (!response.isSuccessful()) {
                    return ReturnT.fail("HTTP " + response.code());
                }

                String respBody = response.body().string();
                return JSON.parseObject(respBody, new TypeReference<ReturnT<String>>(){});
            }
        } catch (SocketTimeoutException e) {
            logger.error("Trigger timeout, address:{}, jobId:{}", address, triggerParam.getJobId());
            return ReturnT.fail("TRIGGER_TIMEOUT");
        } catch (IOException e) {
            logger.error("Trigger error, address:{}, jobId:{}", address, triggerParam.getJobId(), e);
            return ReturnT.fail("TRIGGER_FAIL:" + e.getMessage());
        }
    }

    /**
     * 生成请求签名
     */
    private String generateToken(TriggerParam param) {
        String data = param.getJobId() + "|" + param.getLogId() + "|" + System.currentTimeMillis();
        return HmacUtils.hmacSha256Hex(accessKeySecret, data);
    }
}
```

### 3.2 Netty实现内嵌HTTP服务器

```java
/**
 * 基于Netty的内嵌HTTP服务器
 * 用于执行器接收调度请求
 */
public class EmbedHttpServer {

    private EventLoopGroup bossGroup;
    private EventLoopGroup workerGroup;
    private Channel serverChannel;
    private final int port;
    private final ExecutorBiz executorBiz;

    public EmbedHttpServer(int port, ExecutorBiz executorBiz) {
        this.port = port;
        this.executorBiz = executorBiz;
    }

    public void start() throws InterruptedException {
        bossGroup = new NioEventLoopGroup(1);  // 单线程处理accept
        workerGroup = new NioEventLoopGroup();  // 默认CPU核数*2

        ServerBootstrap bootstrap = new ServerBootstrap();
        bootstrap.group(bossGroup, workerGroup)
            .channel(NioServerSocketChannel.class)
            .childHandler(new ChannelInitializer<SocketChannel>() {
                @Override
                protected void initChannel(SocketChannel ch) {
                    ChannelPipeline pipeline = ch.pipeline();

                    // HTTP编解码器
                    pipeline.addLast(new HttpServerCodec());

                    // 聚合HTTP消息 (最大1MB)
                    pipeline.addLast(new HttpObjectAggregator(1024 * 1024));

                    // 空闲检测 (60秒无数据断开)
                    pipeline.addLast(new IdleStateHandler(0, 0, 60));

                    // 业务处理器
                    pipeline.addLast(new EmbedHttpServerHandler(executorBiz));
                }
            })
            .option(ChannelOption.SO_BACKLOG, 128)
            .childOption(ChannelOption.SO_KEEPALIVE, true);

        serverChannel = bootstrap.bind(port).sync().channel();
        logger.info("Embed HTTP server started on port {}", port);
    }

    public void stop() {
        if (serverChannel != null) {
            serverChannel.close();
        }
        if (bossGroup != null) {
            bossGroup.shutdownGracefully();
        }
        if (workerGroup != null) {
            workerGroup.shutdownGracefully();
        }
    }
}

/**
 * HTTP请求处理器
 */
@ChannelHandler.Sharable
public class EmbedHttpServerHandler extends SimpleChannelInboundHandler<FullHttpRequest> {

    private final ExecutorBiz executorBiz;

    @Override
    protected void channelRead0(ChannelHandlerContext ctx, FullHttpRequest request) {
        // 解析URI
        String uri = request.uri();
        HttpMethod method = request.method();

        // 读取请求体
        ByteBuf content = request.content();
        byte[] body = new byte[content.readableBytes()];
        content.readBytes(body);
        String requestData = new String(body, StandardCharsets.UTF_8);

        // 路由处理
        ReturnT<String> result;
        try {
            switch (uri) {
                case "/run":
                    TriggerParam triggerParam = JSON.parseObject(requestData, TriggerParam.class);
                    result = executorBiz.run(triggerParam);
                    break;
                case "/kill":
                    KillParam killParam = JSON.parseObject(requestData, KillParam.class);
                    result = executorBiz.kill(killParam);
                    break;
                case "/log":
                    LogParam logParam = JSON.parseObject(requestData, LogParam.class);
                    result = executorBiz.log(logParam);
                    break;
                case "/beat":
                    result = ReturnT.success();
                    break;
                default:
                    result = ReturnT.fail("Invalid request path: " + uri);
            }
        } catch (Exception e) {
            logger.error("Request processing error, uri:{}", uri, e);
            result = ReturnT.fail("System error: " + e.getMessage());
        }

        // 构造响应
        String responseJson = JSON.toJSONString(result);
        FullHttpResponse response = new DefaultFullHttpResponse(
            HttpVersion.HTTP_1_1,
            HttpResponseStatus.OK,
            Unpooled.copiedBuffer(responseJson, StandardCharsets.UTF_8)
        );
        response.headers()
            .set(HttpHeaderNames.CONTENT_TYPE, "application/json; charset=UTF-8")
            .set(HttpHeaderNames.CONTENT_LENGTH, response.content().readableBytes());

        ctx.writeAndFlush(response).addListener(ChannelFutureListener.CLOSE);
    }

    @Override
    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
        logger.error("Channel exception", cause);
        ctx.close();
    }
}
```

## 5. 分布式协调

### 5.1 基于数据库的分布式锁

```java
/**
 * 基于数据库的分布式锁
 * 适用于调度中心集群选主场景
 */
@Component
public class DbDistributedLock {

    @Autowired
    private JdbcTemplate jdbcTemplate;

    /**
     * 尝试获取锁
     * @param lockName 锁名称
     * @param expireSeconds 锁过期时间（防止死锁）
     * @return 是否获取成功
     */
    public boolean tryLock(String lockName, int expireSeconds) {
        try {
            String owner = getServerId();
            int updated = jdbcTemplate.update(
                "INSERT INTO job_lock (lock_name, owner, expire_time) VALUES (?, ?, DATE_ADD(NOW(), INTERVAL ? SECOND)) " +
                "ON DUPLICATE KEY UPDATE " +
                "owner = IF(expire_time < NOW(), VALUES(owner), owner), " +
                "expire_time = IF(expire_time < NOW(), VALUES(expire_time), expire_time), " +
                "owner = IF(owner = ?, VALUES(owner), owner)",
                lockName, owner, expireSeconds, owner
            );

            if (updated > 0) {
                // 再次确认是否是自己持有的锁
                String currentOwner = jdbcTemplate.queryForObject(
                    "SELECT owner FROM job_lock WHERE lock_name = ?",
                    String.class, lockName
                );
                return owner.equals(currentOwner);
            }
            return false;
        } catch (Exception e) {
            logger.error("Try lock failed, lockName:{}", lockName, e);
            return false;
        }
    }

    /**
     * 释放锁
     */
    public void unlock(String lockName) {
        try {
            String owner = getServerId();
            jdbcTemplate.update(
                "DELETE FROM job_lock WHERE lock_name = ? AND owner = ?",
                lockName, owner
            );
        } catch (Exception e) {
            logger.error("Unlock failed, lockName:{}", lockName, e);
        }
    }

    /**
     * 续约锁（续期）
     */
    public boolean renewLock(String lockName, int expireSeconds) {
        try {
            String owner = getServerId();
            int updated = jdbcTemplate.update(
                "UPDATE job_lock SET expire_time = DATE_ADD(NOW(), INTERVAL ? SECOND) " +
                "WHERE lock_name = ? AND owner = ?",
                expireSeconds, lockName, owner
            );
            return updated > 0;
        } catch (Exception e) {
            logger.error("Renew lock failed, lockName:{}", lockName, e);
            return false;
        }
    }

    private String getServerId() {
        return IpUtil.getIp() + ":" + System.currentTimeMillis();
    }
}
```

```java
@Component
@ConditionalOnProperty(name = "zjob.lock.type", havingValue = "redis")
public class RedisDistributedLock {

    @Autowired
    private RedissonClient redissonClient;

    /**
     * 可重入锁，支持看门狗自动续期
     */
    public boolean tryLock(String lockName, long waitTime, long leaseTime) {
        RLock lock = redissonClient.getLock("zjob:lock:" + lockName);
        try {
            return lock.tryLock(waitTime, leaseTime, TimeUnit.SECONDS);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            return false;
        }
    }

     * 公平锁 - 按请求顺序获取锁
     */
    public boolean tryFairLock(String lockName, long waitTime) {

        try {
            return fairLock.tryLock(waitTime, 30, TimeUnit.SECONDS);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            return false;
        }
    }

     * 红锁 - 多Redis节点保证高可用
```

```java
/**
 * 路由策略/负载均衡算法实现
 */
public class RouterStrategy {

    /**
     * 轮询 (Round Robin)
     * 按顺序轮流选择，公平分配
     */
    public static class RoundRobin implements ExecutorRouter {
        private final AtomicInteger counter = new AtomicInteger(0);

        @Override
        public String route(List<String> addressList) {
            if (addressList == null || addressList.isEmpty()) {
                return null;
            }
            int index = counter.getAndIncrement() % addressList.size();
            return addressList.get(Math.abs(index));
        }
    }

    /**
     * 随机 (Random)
     * 随机选择，分布均匀
     */
    public static class Random implements ExecutorRouter {
        private final ThreadLocalRandom random = ThreadLocalRandom.current();

        @Override
        public String route(List<String> addressList) {
            if (addressList == null || addressList.isEmpty()) {
                return null;
            }
            int index = random.nextInt(addressList.size());
            return addressList.get(index);
        }
    }

    /**
     * 一致性哈希 (Consistent Hash)
     * 相同参数路由到相同节点，适合有状态任务
     */
    public static class ConsistentHash implements ExecutorRouter {
        private static final int VIRTUAL_NODES = 150; // 虚拟节点数

        @Override
        public String route(List<String> addressList, String jobId) {
            if (addressList == null || addressList.isEmpty()) {
                return null;
            }

            // 构建哈希环
            TreeMap<Long, String> hashRing = new TreeMap<>();
            for (String address : addressList) {
                for (int i = 0; i < VIRTUAL_NODES; i++) {
                    long hash = hash(address + "#" + i);
                    hashRing.put(hash, address);
                }
            }

            // 计算jobId的哈希值，找到最近的节点
            long jobHash = hash(jobId);
            SortedMap<Long, String> tailMap = hashRing.tailMap(jobHash);
            Long targetKey = tailMap.isEmpty() ? hashRing.firstKey() : tailMap.firstKey();
            return hashRing.get(targetKey);
        }

        // FNV1_32_HASH算法
        private long hash(String key) {
            final int p = 16777619;
            int hash = (int) 2166136261L;
            for (int i = 0; i < key.length(); i++) {
                hash = (hash ^ key.charAt(i)) * p;
            }
            hash += hash << 13;
            hash ^= hash >> 7;
            hash += hash << 3;
            hash ^= hash >> 17;
            hash += hash << 5;
            return hash & 0x7FFFFFFF;
        }
    }

    /**
     * 最近最少使用 (LRU)
     * 选择调用次数最少的节点
     */
    public static class LRU implements ExecutorRouter {
        private final ConcurrentHashMap<String, AtomicInteger> callCounts = new ConcurrentHashMap<>();

        @Override
        public String route(List<String> addressList) {
            if (addressList == null || addressList.isEmpty()) {
                return null;
            }

            // 找到调用次数最少的地址
            String selected = null;
            int minCount = Integer.MAX_VALUE;

            for (String address : addressList) {
                int count = callCounts.computeIfAbsent(address, k -> new AtomicInteger(0)).get();
                if (count < minCount) {
                    minCount = count;
                    selected = address;
                }
            }

            // 增加计数
            if (selected != null) {
                callCounts.get(selected).incrementAndGet();
            }

            return selected;
        }
    }
}
```

## 9. 容错与重试

```java
/**
 * 重试机制实现
 */
@Component
public class RetryTemplate {

    /**
     * 带重试的执行
     * @param operation 业务操作
     * @param retryTimes 重试次数
     * @param backoffPolicy 退避策略
     * @param retryableExceptions 允许重试的异常类型
     */
    public <T> T executeWithRetry(
            Callable<T> operation,
            int retryTimes,
            BackoffPolicy backoffPolicy,
            Class<? extends Exception>[] retryableExceptions) throws Exception {

        int attempts = 0;
        Exception lastException = null;

        while (attempts <= retryTimes) {
            try {
                return operation.call();
            } catch (Exception e) {
                lastException = e;

                // 判断是否允许重试
                if (!shouldRetry(e, retryableExceptions)) {
                    throw e;
                }

                attempts++;
                if (attempts > retryTimes) {
                    break;
                }

                // 计算退避时间并等待
                long backoffTime = backoffPolicy.getBackoffTime(attempts);
                logger.warn("Operation failed, retrying in {}ms, attempt {}/{}",
                    backoffTime, attempts, retryTimes);
                Thread.sleep(backoffTime);
            }
        }

        throw new RetryExhaustedException("All retry attempts exhausted", lastException);
    }

    private boolean shouldRetry(Exception e, Class<? extends Exception>[] retryableExceptions) {
        if (retryableExceptions == null || retryableExceptions.length == 0) {
            return true; // 默认所有异常都重试
        }
        for (Class<? extends Exception> clazz : retryableExceptions) {
            if (clazz.isInstance(e)) {
                return true;
            }
        }
        return false;
    }

    /**
     * 退避策略接口
     */
    public interface BackoffPolicy {
        long getBackoffTime(int attempt);
    }

    /**
     * 固定间隔退避
     */
    public static class FixedBackoff implements BackoffPolicy {
        private final long interval;

        public FixedBackoff(long interval, TimeUnit unit) {
            this.interval = unit.toMillis(interval);
        }

        @Override
        public long getBackoffTime(int attempt) {
            return interval;
        }
    }

    /**
     * 指数退避 (2^n)
     */
    public static class ExponentialBackoff implements BackoffPolicy {
        private final long initialInterval;
        private final long maxInterval;
        private final double multiplier;

        public ExponentialBackoff(long initialInterval, long maxInterval, double multiplier) {
            this.initialInterval = initialInterval;
            this.maxInterval = maxInterval;
            this.multiplier = multiplier;
        }

        @Override
        public long getBackoffTime(int attempt) {
            long interval = (long) (initialInterval * Math.pow(multiplier, attempt - 1));
            return Math.min(interval, maxInterval);
        }
    }
}
```

### 11.1 请求签名验证

```java
/**
 * API请求签名验证
 * 防止请求被篡改和重放攻击
 */
@Component
public class ApiSignatureValidator {

    @Value("${zjob.access.key:}")
    private String accessKey;

    @Value("${zjob.access.secret:}")
    private String accessSecret;

    /**
     * 验证请求签名
     * @param request 请求对象
     * @return 是否验证通过
     */
    public boolean validate(HttpServletRequest request) {
        String accessKey = request.getHeader("X-Access-Key");
        String timestamp = request.getHeader("X-Timestamp");
        String nonce = request.getHeader("X-Nonce");
        String signature = request.getHeader("X-Signature");

        // 1. 检查必填参数
        if (StringUtils.isAnyBlank(accessKey, timestamp, nonce, signature)) {
            logger.warn("Missing required signature headers");
            return false;
        }

        // 2. 验证AccessKey
        if (!this.accessKey.equals(accessKey)) {
            logger.warn("Invalid access key: {}", accessKey);
            return false;
        }

        // 3. 验证时间戳 (5分钟有效期，防止重放攻击)
        long requestTime = Long.parseLong(timestamp);
        long currentTime = System.currentTimeMillis();
        if (Math.abs(currentTime - requestTime) > 5 * 60 * 1000) {
            logger.warn("Request timestamp expired");
            return false;
        }

        // 4. 验证nonce (可选，配合Redis防止重复请求)
        if (!validateNonce(nonce)) {
            logger.warn("Duplicate nonce detected");
            return false;
        }

        // 5. 验证签名
        String expectedSignature = generateSignature(request, timestamp, nonce);
        if (!signature.equalsIgnoreCase(expectedSignature)) {
            logger.warn("Signature mismatch");
            return false;
        }

        return true;
    }

    /**
     * 生成请求签名
     * 签名算法: HMAC-SHA256(accessSecret, method + url + timestamp + nonce + body)
     */
    public String generateSignature(HttpServletRequest request, String timestamp, String nonce) {
        try {
            String method = request.getMethod().toUpperCase();
            String url = request.getRequestURI();
            String body = getRequestBody(request);

            // 构造签名串
            String signContent = String.format("%s|%s|%s|%s|%s",
                method, url, timestamp, nonce, body);

            // HMAC-SHA256签名
            Mac mac = Mac.getInstance("HmacSHA256");
            SecretKeySpec secretKeySpec = new SecretKeySpec(accessSecret.getBytes(StandardCharsets.UTF_8), "HmacSHA256");
            mac.init(secretKeySpec);
            byte[] hash = mac.doFinal(signContent.getBytes(StandardCharsets.UTF_8));

            return Base64.getEncoder().encodeToString(hash);
        } catch (Exception e) {
            throw new RuntimeException("Generate signature failed", e);
        }
    }

    private String getRequestBody(HttpServletRequest request) {
        try {
            BufferedReader reader = request.getReader();
            return reader.lines().collect(Collectors.joining(System.lineSeparator()));
        } catch (IOException e) {
            return "";
        }
    }

    private boolean validateNonce(String nonce) {
        // 使用Redis存储nonce，设置5分钟过期
        String key = "zjob:nonce:" + nonce;
        Boolean success = redisTemplate.opsForValue().setIfAbsent(key, "1", 5, TimeUnit.MINUTES);
        return Boolean.TRUE.equals(success);
    }
}
```

### 12.1 多级缓存架构

```java
/**
 * 多级缓存实现
 * L1: Caffeine本地缓存 (微秒级)
 * L2: Redis分布式缓存 (毫秒级)
 * L3: 数据库 (十毫秒级)
 */
@Component
public class MultiLevelCache<K, V> {

    // L1: 本地缓存
    private final Cache<K, V> localCache;

    // L2: Redis缓存
    private final StringRedisTemplate redisTemplate;

    private final ObjectMapper objectMapper;
    private final Class<V> valueType;
    private final String cacheName;

    public MultiLevelCache(
            String cacheName,
            Class<V> valueType,
            StringRedisTemplate redisTemplate,
            int localExpireSeconds,
            int localMaxSize) {

        this.cacheName = cacheName;
        this.valueType = valueType;
        this.redisTemplate = redisTemplate;
        this.objectMapper = new ObjectMapper();

        // 初始化Caffeine本地缓存
        this.localCache = Caffeine.newBuilder()
            .maximumSize(localMaxSize)
            .expireAfterWrite(localExpireSeconds, TimeUnit.SECONDS)
            .recordStats()  // 开启统计
            .removalListener((key, value, cause) -> {
                logger.debug("Local cache entry removed: key={}, cause={}", key, cause);
            })
            .build();
    }

    /**
     * 获取缓存值
     */
    public V get(K key, Callable<V> loader) {
        // 1. 查询L1本地缓存
        V value = localCache.getIfPresent(key);
        if (value != null) {
            Metrics.counter("cache.l1.hit", cacheName).increment();
            return value;
        }

        // 2. 查询L2 Redis缓存
        String redisKey = buildRedisKey(key);
        String jsonValue = redisTemplate.opsForValue().get(redisKey);
        if (StringUtils.isNotBlank(jsonValue)) {
            try {
                value = objectMapper.readValue(jsonValue, valueType);
                // 回填L1缓存
                localCache.put(key, value);
                Metrics.counter("cache.l2.hit", cacheName).increment();
                return value;
            } catch (JsonProcessingException e) {
                logger.error("Deserialize cache value failed", e);
            }
        }

        // 3. 回源到数据库
        Metrics.counter("cache.miss", cacheName).increment();
        try {
            value = loader.call();
            if (value != null) {
                // 写入L1和L2
                put(key, value);
            }
            return value;
        } catch (Exception e) {
            throw new CacheLoadException("Load from source failed", e);
        }
    }

    /**
     * 写入缓存
     */
    public void put(K key, V value) {
        // 1. 写入L1
        localCache.put(key, value);

        // 2. 写入L2
        try {
            String redisKey = buildRedisKey(key);
            String jsonValue = objectMapper.writeValueAsString(value);
            redisTemplate.opsForValue().set(redisKey, jsonValue,
                getL2ExpireSeconds(), TimeUnit.SECONDS);
        } catch (JsonProcessingException e) {
            logger.error("Serialize cache value failed", e);
        }
    }

    /**
     * 删除缓存
     */
    public void evict(K key) {
        // 1. 删除L1
        localCache.invalidate(key);

        // 2. 删除L2
        String redisKey = buildRedisKey(key);
        redisTemplate.delete(redisKey);

        // 3. 发送缓存失效广播 (用于集群环境)
        publishEvictEvent(key);
    }

    /**
     * 清空所有缓存
     */
    public void clear() {
        localCache.invalidateAll();
        Set<String> keys = redisTemplate.keys(cacheName + ":*");
        if (!CollectionUtils.isEmpty(keys)) {
            redisTemplate.delete(keys);
        }
    }

    /**
     * 获取缓存统计
     */
    public CacheStats getStats() {
        com.github.benmanes.caffeine.cache.CacheStats localStats = localCache.stats();

        return CacheStats.builder()
            .cacheName(cacheName)
            .localSize(localCache.estimatedSize())
            .localHitCount(localStats.hitCount())
            .localMissCount(localStats.missCount())
            .localHitRate(localStats.hitRate())
            .localLoadCount(localStats.loadCount())
            .localEvictionCount(localStats.evictionCount())
            .localAverageLoadPenalty(localStats.averageLoadPenalty())
            .build();
    }

    private String buildRedisKey(K key) {
        return cacheName + ":" + key.toString();
    }

    private void publishEvictEvent(K key) {
        // 发布到Redis频道，通知其他节点失效本地缓存
        String channel = "zjob:cache:evict:" + cacheName;
        redisTemplate.convertAndSend(channel, key.toString());
    }

    private int getL2ExpireSeconds() {
        // L2过期时间通常是L1的2-3倍
        return 180; // 3分钟
    }
}
```
